const [validity, applyOn] = createValidity({
  type: "longitude",
  step: 0.1,
});

const run = (value) => {
  applyOn(value);
  return structuredClone(validity);
};

return {
  "3.000001": run("3.000001"),
  "2.67": run("2.67"),
  "-179.95": run("-179.95"),
  "3.05": run("3.05"),
};
