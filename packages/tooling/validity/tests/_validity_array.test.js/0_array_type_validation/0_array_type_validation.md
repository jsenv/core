# [array type validation](../../validity_array.test.js)

```js
const [validity, applyOn] = createValidity({
"type": "array"
});

const cases = [[1,2,3],[],[62],["true"],"{key:\"value\"}","[1,2,3]","not an array",undefined];
const rows = cases.map((value) => {
  applyOn(value);
  return [
    cell(humanize(value)),
    cell(humanize(validity.value)),
    cell(humanize(validity.valid)),
    cell(humanize(validity.representations.valid?.value)),
    cell(humanize(validity.type)),
  ];
});

return renderTable(
  [
    [
      cell("input"),
      cell(".value"),
      cell(".valid"),
      cell(".representations.valid.value"),
      cell(".type"),
    ],
    ...rows,
  ],
  { borderCollapse: true },
);
```
