# [number step](../../validity_number.test.js)

```js
const [validity, applyOn] = createValidity({
  type: "number",
  step: 0.1,
});

const cases = [1.2, 1.23, 3.000001, 3.05, 2.67];
const rows = cases.map((value) => {
  applyOn(value);
  return [
    cell(humanize(value)),
    cell(humanize(validity.valid)),
    cell(humanize(validity.representations.valid?.value)),
    cell(validity.step ?? "-"),
  ];
});

return renderTable(
  [
    [
      cell("input"),
      cell(".valid"),
      cell(".representations.valid.value"),
      cell(".step"),
    ],
    ...rows,
  ],
  { borderCollapse: true },
);
```
