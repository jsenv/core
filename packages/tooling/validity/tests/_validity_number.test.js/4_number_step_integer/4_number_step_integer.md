# [number step integer](../../validity_number.test.js)

```js
const [validity, applyOn] = createValidity({
  type: "number",
  step: 1,
});

const cases = [5, 5.5];
const rows = cases.map((value) => {
  applyOn(value);
  return [
    cell(humanize(value)),
    cell(humanize(validity.valid)),
    cell(humanize(validity.representations.valid?.value)),
    cell(validity.step ?? "-"),
  ];
});

return renderTable(
  [
    [
      cell("input"),
      cell(".valid"),
      cell(".representations.valid.value"),
      cell(".step"),
    ],
    ...rows,
  ],
  { borderCollapse: true },
);
```
