# [number combined min max step](../../validity_number.test.js)

```js
const [validity, applyOn] = createValidity({
  type: "number",
  min: 0,
  max: 10,
  step: 0.5,
});

const cases = [5.5, -2.3];
const rows = cases.map((value) => {
  applyOn(value);
  return [
    cell(humanize(value)),
    cell(validity.valid ? "✓" : "✗"),
    cell(validity.representations.valid?.value ?? "-"),
    cell(
      [validity.type, validity.min, validity.max, validity.step]
        .filter(Boolean)
        .join(", ") || "-",
    ),
  ];
});

return renderTable(
  [
    [
      cell("value"),
      cell("valid"),
      cell("valid suggestion"),
      cell("errors"),
    ],
    ...rows,
  ],
  { borderCollapse: true },
);
```
