# [number combined min max step](../../validity_number.test.js)

```js
const [validity, applyOn] = createValidity({
  type: "number",
  min: 0,
  max: 10,
  step: 0.5,
});

const cases = [5.5, -2.3];
const rows = cases.map((value) => {
  applyOn(value);
  return [
    cell(humanize(value)),
    cell(humanize(validity.valid)),
    cell(humanize(validity.representations.valid?.value)),
    cell(
      [validity.type, validity.min, validity.max, validity.step]
        .filter(Boolean)
        .join(", ") || "-",
    ),
  ];
});

return renderTable(
  [
    [
      cell("input"),
      cell(".valid"),
      cell(".representations.valid.value"),
      cell("errors"),
    ],
    ...rows,
  ],
  { borderCollapse: true },
);
```
