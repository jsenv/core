# [time type validation](../../validity_time.test.js)

```js
const [validity, applyOn] = createValidity({
  type: "time",
});

const cases = [
  "14:30",
  "09:15:30",
  "23:59:59",
  "24:00",
  "14:60",
  undefined,
];
const rows = cases.map((value) => {
  applyOn(value);
  return [
    cell(humanize(value)),
    cell(humanize(validity.value)),
    cell(humanize(validity.valid)),
    cell(humanize(validity.representations.valid?.value)),
    cell(humanize(validity.type)),
  ];
});

return renderTable(
  [
    [
      cell("input"),
      cell(".value"),
      cell(".valid"),
      cell(".representations.valid.value"),
      cell(".type"),
    ],
    ...rows,
  ],
  { borderCollapse: true },
);
```
