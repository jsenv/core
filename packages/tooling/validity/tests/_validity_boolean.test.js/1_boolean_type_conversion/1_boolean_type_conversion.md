# [boolean type conversion](../../validity_boolean.test.js)

```js
const [validity, applyOn] = createValidity({
  type: "boolean",
});

const cases = [
  true,
  false,
  "true",
  "false",
  "on",
  "1",
  1,
  0,
  "toto",
  undefined,
];

const grid = [
  [
    cell("input"),
    cell(".valid"),
    cell(".representations.valid.value"),
    cell(".type"),
  ],
  ...cases.map((value) => {
    applyOn(value);
    return [
      cell(humanize(value)),
      cell(humanize(validity.valid)),
      cell(humanize(validity.representations.valid?.value)),
      cell(humanize(validity.type)),
    ];
  }),
];

return renderTable(grid, { borderCollapse: true });
```
