const cases = [
  // strings
  "2hour",
  "2hour15minute",
  "1year",
  "30second",
  "1.14second",
  "-1second",
  "1second140millisecond",
  "30",
  "",
  null,
  // objects
  { hours: 2, minutes: 15 },
  { years: 1 },
  { minutes: 30 },
  { seconds: "-1" },
  { hours: "2a" },
  {},
];

const rows = cases.map((value) => {
  const result = durationToSeconds(value);
  return [cell(humanize(value)), cell(humanize(result))];
});

return renderTable(
  [[cell("input"), cell("durationToSeconds()")], ...rows],
  { borderCollapse: true },
);
