const [validity, applyOn] = createValidity({
  type: "boolean",
});

const run = (value) => {
  applyOn(value);
  return structuredClone(validity);
};

return {
  "true": run(true),
  '"true"': run("true"),
  '"false"': run("false"),
  "1": run(1),
};
