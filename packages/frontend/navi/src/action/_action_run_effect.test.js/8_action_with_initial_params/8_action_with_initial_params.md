# [action with initial params](../../action_run_effect.test.js)

```js
const paramsSignal = signal();
const runCalls = [];
const action = createAction(
  (p) => {
    runCalls.push(p);
  },
  { params: { foo: true } },
);
actionRunEffect(action, () => paramsSignal.value);
paramsSignal.value = { bar: true };

return runCalls;
```
