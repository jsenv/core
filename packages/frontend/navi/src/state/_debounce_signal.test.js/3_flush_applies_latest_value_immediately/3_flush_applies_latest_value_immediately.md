# [flush applies latest value immediately](../../debounce_signal.test.js)

```js
const delay = 50;
const src = signal("a");
const debounced = debounceSignal(src, { delay });

src.value = "b";
const valueBeforeFlush = debounced.value;
debounced.flush();
const valueAfterFlush = debounced.value;

return { valueBeforeFlush, valueAfterFlush };
```

{
  "valueBeforeFlush": "a",
  "valueAfterFlush": "b"
}
