# [throws when syncResourceToSignals is called on a scoped resource](../../sync_owned_resource_to_signals.test.js)

```js
const TABLE = resource("table", {
  idKey: "tableoid",
  uniqueKeys: ["tablename"],
});
const TABLE_COLUMN = TABLE.scopedMany("columns", {
  idKey: "column_name",
});

const someSignal = signal("value");
let error = null;
try {
  syncResourceToSignals(TABLE_COLUMN, { column_name: someSignal });
} catch (e) {
  error = e.message;
}

return { error };
```

```js
{
  "error": 'syncResourceToSignals: "table.columns" is a scoped resource (scopedMany/scopedOne). Use syncOwnedResourceToSignals instead.'
}
```
