# [GET is reset when DELETE targets its id](../../resource_graph_lifecycle.test.js)

```js
const USER = resource("user", {
  GET: async ({ id }) => ({ id, name: "Alice" }),
  DELETE: async ({ id }) => id,
});

await USER.GET({ id: 1 });
const stateAfterGet = {
  completed: USER.GET.bindParams({ id: 1 }).completed,
};

await USER.DELETE({ id: 1 });
await waitForRerun();

const stateAfterDelete = {
  completed: USER.GET.bindParams({ id: 1 }).completed,
};

return { stateAfterGet, stateAfterDelete };
```

```js
{
  "stateAfterGet": {
    "completed": true
  },
  "stateAfterDelete": {
    "completed": false
  }
}
```
