# [GET_MANY does NOT rerun when PUT is called (default rerunOn)](../../resource_graph_lifecycle.test.js)

```js
let getManyCallCount = 0;
const USER = resource("user", {
  GET_MANY: async () => {
    getManyCallCount++;
    return [{ id: 1, name: "Alice" }];
  },
  PUT: async ({ id, name }) => ({ id, name }),
});

await USER.GET_MANY.run();
await USER.PUT({ id: 1, name: "Alice Updated" });
await waitForRerun();

return { getManyCallCount };
```
