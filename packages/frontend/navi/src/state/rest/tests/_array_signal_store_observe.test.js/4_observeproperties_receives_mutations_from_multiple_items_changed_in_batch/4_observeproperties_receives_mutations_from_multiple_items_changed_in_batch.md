# [observeProperties receives mutations from multiple items changed in batch](../../array_signal_store_observe.test.js)

```js
const store = arraySignalStore([], "id", {});
store.upsert([
  { id: 1, name: "alice" },
  { id: 2, name: "bob" },
]);

const calls = [];
store.observeProperties((mutationsArray) => {
  calls.push(mutationsArray);
});

store.upsert([
  { id: 1, name: "alice-renamed" },
  { id: 2, name: "bob-renamed" },
]);

return { calls };
```
