# [observeProperties with properties filter not called when unrelated property changes](../../array_signal_store_observe.test.js)

```js
const store = arraySignalStore([], "id", {});
store.upsert({ id: 1, name: "alice", age: 30 });

let called = false;
store.observeProperties(
  () => {
    called = true;
  },
  { properties: ["name"] },
);

store.upsert({ id: 1, age: 31 }); // only age changes, name not watched

return { called };
```

```js
{
  "called": false
}
```
